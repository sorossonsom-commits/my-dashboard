    function updateMainGauge(elementId, value, maxVal) {

    const API_URL = "https://script.google.com/macros/s/AKfycbybr4Gh356bvITe0PR_dAGVurU1czBf1Zd0v5c2dQsV9ijG2GiZh2IXryGt96W-cBS1/exec"; 

    function loadData() {
        if (!selectedStation) return;
        showLoading(true);

        const requestUrl = API_URL + "?action=getData&station=" + encodeURIComponent(selectedStation);

        fetch(requestUrl)
            .then(response => response.json())
            .then(dataArray => {
                if(dataArray.error) throw new Error(dataArray.error);
                renderDashboard(dataArray); // ส่งข้อมูลไปโชว์หน้าเว็บ
                showLoading(false);
            })
            .catch(error => {
                console.error("Error:", error);
                showLoading(false);
                alert("ดึงข้อมูลไม่ได้: " + error.message);
            });
    }
    function renderDashboard(dataArray) {
        var btn = document.getElementById('refreshBtn');
        var status = document.getElementById('status');
        
        if (dataArray && dataArray.length > 0) {
            const latestRow = dataArray[0];
            
            // ดึงค่า System Status
            let timeStr = latestRow[0] || '---';
            let wifiStr = latestRow[53] || '---';
            let bleStr = latestRow[54] || '---';

            document.getElementById('lastUpdateStatus').innerText = timeStr;
            document.getElementById('wifiStatus').innerText = wifiStr;
            document.getElementById('bleStatus').innerText = bleStr;

            // ดึงค่าอื่นๆ
            let totalVoltageStr = latestRow[25] || '0'; 
            let currentStr = latestRow[26] || '0';
            let temp1Str = latestRow[51] || '---'; 
            let temp2Str = latestRow[52] || '---'; 
            let extCurrentStr = latestRow[55] || '---'; 
            let current3Str = latestRow[56] || '---';
            let current4Str = latestRow[57] || '---'; 

            const voltage = parseFloat(totalVoltageStr.replace(',', '.')); 
            const c1 = parseFloat(currentStr.replace(',', '.'));
            const c2 = parseFloat(extCurrentStr.replace(',', '.')); 
            const c3 = parseFloat(current3Str.replace(',', '.')); 
            const c4 = parseFloat(current4Str.replace(',', '.'));
            
            let power = (!isNaN(voltage) && !isNaN(c1)) ? (voltage * c1).toFixed(2) : 'N/A'; 
            let power2 = (!isNaN(voltage) && !isNaN(c2)) ? (voltage * c2).toFixed(2) : 'N/A';
            
            document.getElementById('latestTotalValue').innerText = totalVoltageStr;
            document.getElementById('latestCurrentValue').innerText = currentStr;
            document.getElementById('latestExtValue').innerText = extCurrentStr;
            document.getElementById('latestCurrent3Value').innerText = current3Str;
            document.getElementById('latestCurrent4Value').innerText = current4Str;
            document.getElementById('latestPowerValue').innerText = power;
            document.getElementById('latestPowerValue2').innerText = power2;
            document.getElementById('latestTempValue1').innerText = temp1Str;
            document.getElementById('latestTempValue2').innerText = temp2Str;

            // อัปเดตเข็มไมล์หลักทั้ง 5 ช่อง
            updateMainGauge('gaugeFillVoltage', Math.abs(voltage), MAX_SYS_VOLTAGE);
            updateMainGauge('gaugeFillCurrent1', Math.abs(c1), MAX_SYS_CURRENT);
            updateMainGauge('gaugeFillExt', Math.abs(c2), MAX_SYS_CURRENT);
            updateMainGauge('gaugeFillCurrent3', Math.abs(c3), MAX_SYS_CURRENT);

                let resDataIndex = 26 + i;

                document.getElementById(`cellValue${i}`).innerText = cellVoltage.toFixed(3) + ' V';
                let resEl = document.getElementById(`resValue${i}`);
                if (resEl) resEl.innerText = (resValueStr === '---' || resValueStr === '') ? '---' : resValueStr + ' mΩ';
                
                let gauge = document.getElementById(`gaugeFill${i}`);
                let needle = document.getElementById(`needle${i}`);
                if (gauge && needle) {

                    gauge.style.strokeDashoffset = SEMI_CIRCUMFERENCE * (1 - (pct / 100));
                    
                    let gaugeColor = 'var(--accent-cyan)';

                    COLOR_THRESHOLDS.forEach(t => {
                        if (cellVoltage >= t.value) {

    function getPath(n1, n2) {
        let p1 = { x: (n1.x / 100) * fw, y: (n1.y / 100) * fh };
        let p2 = { x: (n2.x / 100) * fw, y: (n2.y / 100) * fh };
        let mid = Math.random() > 0.5 ? { x: p1.x, y: p2.y } : { x: p2.x, y: p1.y };
        return [p1, mid, p2];
    }

    function spawnPulse() {
        let targetNode = bmsNodes[Math.floor(Math.random() * (bmsNodes.length - 1)) + 1];
        let path = getPath(targetNode, bmsNodes[0]);
        pulses.push({ path: path, segment: 0, progress: 0, speed: 0.004 + (Math.random() * 0.008) });
    }

    function drawFlowchart() {
        if(document.getElementById('welcomeScreen').style.display === 'none') {
            requestAnimationFrame(drawFlowchart);
            return;
        }
        
        fCtx.clearRect(0, 0, fw, fh);
        fCtx.lineWidth = 1;
        fCtx.strokeStyle = "rgba(0, 242, 254, 0.15)";
        
        for(let i=1; i<bmsNodes.length; i++) {
            let n1 = bmsNodes[0], n2 = bmsNodes[i];
            let p1 = { x: (n1.x / 100) * fw, y: (n1.y / 100) * fh };
            let p2 = { x: (n2.x / 100) * fw, y: (n2.y / 100) * fh };
            fCtx.beginPath();
            fCtx.moveTo(p1.x, p1.y);
            fCtx.lineTo(p1.x, p2.y);
            fCtx.lineTo(p2.x, p2.y);
            fCtx.stroke();
        }

        bmsNodes.forEach(n => {
            let nx = (n.x / 100) * fw, ny = (n.y / 100) * fh;
            fCtx.beginPath();
            fCtx.arc(nx, ny, 6, 0, Math.PI * 2);
            fCtx.fillStyle = "#00f2fe";
            fCtx.fill();
            fCtx.shadowBlur = 10;
            fCtx.shadowColor = "#00f2fe";
            fCtx.fill();
            fCtx.shadowBlur = 0;

            fCtx.font = "bold 14px 'Sarabun', sans-serif";
            fCtx.fillStyle = "#ffffff"; 
            fCtx.textAlign = "center";
            fCtx.textBaseline = "bottom"; 
            fCtx.shadowBlur = 5; 
            fCtx.shadowColor = "#00f2fe";
            fCtx.fillText(n.label, nx, ny - 15); 
            fCtx.shadowBlur = 0;
        });

        if (Math.random() < 0.07 && pulses.length < 25) spawnPulse(); 

        for (let i = pulses.length - 1; i >= 0; i--) {
            let p = pulses[i];
            let pStart = p.path[p.segment];
            let pEnd = p.path[p.segment + 1];
            
            let cx = pStart.x + (pEnd.x - pStart.x) * p.progress;
            let cy = pStart.y + (pEnd.y - pStart.y) * p.progress;
            
            fCtx.beginPath();
            fCtx.arc(cx, cy, 3, 0, Math.PI * 2);
            fCtx.fillStyle = "#ffffff";
            fCtx.shadowBlur = 15;
            fCtx.shadowColor = "#00f2fe";
            fCtx.fill();
            fCtx.shadowBlur = 0;

            p.progress += p.speed;
            if (p.progress >= 1) {
                p.progress = 0;
                p.segment++;
                if (p.segment >= p.path.length - 1) pulses.splice(i, 1);
            }
        }
        requestAnimationFrame(drawFlowchart);
    }
    drawFlowchart();
